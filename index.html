        if(!this.head){
            return undefined
        }

        let pre = this.head
        while( temp.next){
            pre = temp
            temp = temp.next
        }

        this.length--

}

Reverse(){
    let temp = this.head;
    this.head = this.tail;
    this.tail = temp;
    let next = temp.next;
    let prev = null
    for( let i = 0; i < this.length; i++ ){

        temp = next
    }
    return
