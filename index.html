Array.push()

array pushing

                        
const arr = ["A","B","C"];
const arr2 = [1,2,3]
    
// Pushing D onto the end of the array
arr.push("D"); // ["A","B","C","D"]

arr2.push(4,5,6); // [1,2,3,4,5,6]

// You can also use the spread operator
arr.push(...arr2) // ["A","B","C",1,2,3]

Array.pop()

array poping

                        
const arr = ["A","B","C"];
const arr2 = [1,2,3]
    
// Popping removes the last element of an 
// array
arr.pop();  // ["A","B"]

// When popping you can return the popped 
// value
const a = arr.pop()
console.log(a) // "C"

Array.Shift()

array shift

                        
const arr = ["A","B","C"];
    
// shifting removes the first element of

arr.shift(); // ["B","C"]

LinkedList Push()

                    
class LinkedList {
    constructor() {
        this.head = null
        this.tail = null
        this.length = 0
    }
                        
    Push(value){
        const nodeData = new Node(value)
        this.length++;
        if(!this.head){
            this.head = nodeData;
            this.tail = nodeData;
            return;
        }
        else{
                this.tail.next = nodeData;
                this.tail = nodeData;
                return;
        }
    }
}

LinkedList Pop()

                    
class LinkedList {
    constructor() {
        this.head = null
        this.tail = null
        this.length = 0
    }

    Pop(){
        if(!this.head){
            return undefined
        }
        let temp = this.head
        let pre = this.head
        while( temp.next){
            pre = temp
            temp = temp.next
        }
        this.tail = pre
        this.tail.next = null
        this.length--
        if(this.length === 0){
            this.head = null
            this.tail = null
        }
        return temp
    }
