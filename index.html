Array.push()

array pushing

                        
const arr = ["A","B","C"];
const arr2 = [1,2,3]
    
// Pushing D onto the end of the array
arr.push("D");
console.log(arr) // ["A","B","C","D"]

arr2.push(4,5,6);
console.log(arr) // [1,2,3,4,5,6]

// You can also use the spread operator
arr.push(...arr2) // ["A","B","C",1,2,3]

Array.pop()

array poping

                        
const arr = ["A","B","C"];
const arr2 = [1,2,3]
    
// Popping removes the last element of an 
// array
arr.pop();
console.log(arr) // ["A","B"]

// When popping you can return the popped 
// value
const a = arr.pop()
console.log(a) // "C"

Array.Shift()

array shift

                        
const arr = ["A","B","C"];
    
// shifting removes the first element of 
// an arr
arr.shift();
console.log(arr) // ["B","C"]

// When shifting you can return the shifted 
// value
const a = arr.shift()
console.log(a) // "A"

Array.Unshift()

array shift

                        
const arr = ["A","B","C"];
const arr2 = [1,2,3];
                                
// Pushing D onto the end of the array
arr.unshift("D");
console.log(arr) // ["D","A","B","C"]
                            
arr2.unshift(4,5);
console.log(arr) // [4,5,1,2,3]
                            
// You can also use the spread operator
arr.unshift(...arr2) // [1,2,3,"A","B","C"]
