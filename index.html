Insert()

Example:


class LinkedList {
    constructor( ) {
        this.head = null
        this.tail = null
        this.length = 0
    }

    Append(value){
        const nodeData = new Node(value)
        this.length++;
        if(!this.head)
        {
            this.head = nodeData;
            this.tail = nodeData;
            return;
        }
        else{
            this.tail.next = nodeData;
            this.tail = nodeData;
            return;
        }
    }
}

Example:


const List = new LinkedList()

List.Append(5)

console.log(List) /// (5)

List.Append(6)

console.log(List) /// (5)->(6)

InsertAt(value)

Example:


class LinkedList {
constructor( ) {
    this.head = null
    this.tail = null
    this.length = 0
}

InsertAt( value, location ){
    if( location > this.length 
    || location < 0 ){ 
        return undefined;
    } 
    this.length++;
    const nodeData = new Node(value)
    if(location === 0){
     nodeData.next = this.head;
     this.head = nodeData;
     return     
    }
    if(location === this.length - 1){
     this.tail.next = nodeData;
     this.tail = nodeData;
    return  
    }
    let curr = this.head;
    let index = 0
    while((location - 1) > index){
        curr = curr.next
    }
    nodeData.next = curr.next;
    curr.next = nodeData;
}

Example:


const List = new LinkedList()

List.Append(1)

List.Append(2)

console.log(List) /// (1)->(2)

List.InsertAt(5, 0)

console.log(List) /// (5)->(1)->(2)

List.InsertAt(8, 1)

console.log(List) /// (5)->(8)->(1)->(2)
